simple_loop:
    LOAD_CONST 0
    STORE
    JMP .start
    .start:
        LOAD 0
        LOAD_CONST 10
        LT
        EJMPF
        LOAD 0
        LOAD_CONST 1
        ADD
        DUP
        STORE 0
        CALL println 1
        JMP .start
        EJMP
    LOAD_CONST "done"
    CALL println 1
    RET
